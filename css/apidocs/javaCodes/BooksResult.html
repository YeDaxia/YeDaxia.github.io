public class BooksResult implements Serializable{

    private BookResult[] books; //
    private int code; //
    private String message; //

    @JSONField(name="books")
    public BookResult[] getBooks(){
        return books;
    }

    @JSONField(name="books")
    public void setBooks(BookResult[] books){
        this.books = books;
    }

    @JSONField(name="code")
    public int getCode(){
        return code;
    }

    @JSONField(name="code")
    public void setCode(int code){
        this.code = code;
    }

    @JSONField(name="message")
    public String getMessage(){
        return message;
    }

    @JSONField(name="message")
    public void setMessage(String message){
        this.message = message;
    }

}

public class BookResult implements Serializable{

    private String id; //图书id
    private String bookName; //图书名称
    private UserResult author; //图书作者

    @JSONField(name="id")
    public String getId(){
        return id;
    }

    @JSONField(name="id")
    public void setId(String id){
        this.id = id;
    }

    @JSONField(name="bookName")
    public String getBookName(){
        return bookName;
    }

    @JSONField(name="bookName")
    public void setBookName(String bookName){
        this.bookName = bookName;
    }

    @JSONField(name="author")
    public UserResult getAuthor(){
        return author;
    }

    @JSONField(name="author")
    public void setAuthor(UserResult author){
        this.author = author;
    }

}

public class UserResult implements Serializable{

    private String id; //用户id
    private String token; //用户凭证
    private String name; //名字
    private String avatar; //头像

    @JSONField(name="id")
    public String getId(){
        return id;
    }

    @JSONField(name="id")
    public void setId(String id){
        this.id = id;
    }

    @JSONField(name="token")
    public String getToken(){
        return token;
    }

    @JSONField(name="token")
    public void setToken(String token){
        this.token = token;
    }

    @JSONField(name="name")
    public String getName(){
        return name;
    }

    @JSONField(name="name")
    public void setName(String name){
        this.name = name;
    }

    @JSONField(name="avatar")
    public String getAvatar(){
        return avatar;
    }

    @JSONField(name="avatar")
    public void setAvatar(String avatar){
        this.avatar = avatar;
    }

}
